def get_data(data,look_back):
  data_x,data_y=[],[]
  for i in range(len(data)-look_back-1):
    data_x.append(data[i:(i+look_back),0])
    data_y.append(data[i+look_back,0])
  return np.array(data_x),np.array(data_y)
